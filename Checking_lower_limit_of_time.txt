function time_sublimit($k = 0.8) {
    $limit = ini_get('max_execution_time'); 
    $sub_limit = round($limit * $k);
    if($sub_limit === 0) {
        $sub_limit = INF;
    }
    return $sub_limit;
}

    $t = time();

            }

} catch(Exception $e) {
