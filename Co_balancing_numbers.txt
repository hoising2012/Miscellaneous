function Sum_to_n_plus_r($n,$m){
		$Sum="";
        $r=1;
        do{
        	$F = $n+$r;
        	$Sum += $F;
            $r++;
        }while($Sum<$m);
		return $r-1;
}
function Sum_to_n($n){
	$Sum = (($n+1)*$n)/2;
    return $Sum;
}
function condition($n){
	$result = 8*pow($n,2)+8*$n+1;

	$condition;

    	$condition = "Not balancing";
    }
